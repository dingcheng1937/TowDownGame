## 三、代码练习题

### 练习1：数据持久化

```gdscript
const SAVE_PATH = "user://save_data.json"

func save_data():
    var data = {
        "player_hp": player_hp,
        "player_level": player_level,
        "gold": gold
    }
    var file = FileAccess.open(SAVE_PATH, FileAccess.WRITE)
    if file:
        file.store_string(JSON.stringify(data))
        file.close()

func load_data():
    var file = FileAccess.open(SAVE_PATH, FileAccess.READ)
    if file:
        var data = JSON.parse_string(file.get_as_string())
        player_hp = data["player_hp"]
        file.close()
```

### 练习2：成就系统

```gdscript
var achievements = {
    "first_kill": {"name": "初次击杀", "unlocked": false},
    "kill_100": {"name": "百杀斩", "unlocked": false}
}

var kill_count = 0

func add_kill():
    kill_count += 1
    if kill_count == 1 && !achievements["first_kill"]["unlocked"]:
        achievements["first_kill"]["unlocked"] = true
        emit_signal("onAchievementUnlock", "first_kill")
```
